```typescript
// components/TransactionFeed.tsx (Realtime approach)
'use client';
import { createClient } from '@/lib/supabase-browser';
import { useEffect, useState } from 'react';
import type { Transaction } from '@/types/database';

export function TransactionFeed({ orgId }: { orgId: string }) {
  const supabase = createClient();
  const [transactions, setTransactions] = useState<Transaction[]>([]);

  useEffect(() => {
    // Initial load
    supabase.from('transactions')
      .select('*')
      .eq('organization_id', orgId)
      .order('created_at', { ascending: false })
      .limit(50)
      .then(({ data }) => setTransactions(data ?? []));

    // Realtime subscription
    const channel = supabase
      .channel(`org:${orgId}:transactions`)
      .on(
        'postgres_changes',
        { event: 'INSERT', schema: 'public', table: 'transactions', filter: `organization_id=eq.${orgId}` },
        (payload) => setTransactions(prev => [payload.new as Transaction, ...prev])
      )
      .subscribe();

    return () => { supabase.removeChannel(channel); };
  }, [orgId, supabase]);

  return (
    <ul>
      {transactions.map(tx => (
        <li key={tx.id}>
          <strong>@{tx.sender_handle}</strong> → <strong>@{tx.receiver_handle}</strong>
          {' '}{tx.amount} {tx.asset_code}
        </li>
      ))}
    </ul>
  );
}
```

## Phase 4: x402 Premium Features

Phase 4 introduces a revenue layer using the x402 protocol to gate premium SocialPay features behind micropayments.

| Premium Feature | x402 Cost | Notes |
|---|---|---|
| Bulk CSV export of transactions | 1 XLM | Download all transactions as spreadsheet |
| Advanced analytics dashboard | 0.5 XLM/month | Charts, volume trends, top handles |
| Team webhooks | 2 XLM setup | POST to your endpoint on every transaction |
| Custom transaction feed embed | 1 XLM setup | Embed your org feed in Notion or Slack |
| API rate limit increase | 0.1 XLM/day | 10x rate limit for power API users |

```typescript
// Reusable x402 gate middleware
// lib/x402-gate.ts

export function withX402Gate(amountXlm: string) {
  return function gate(handler: (req: Request) => Promise<Response>) {
    return async function (request: Request): Promise<Response> {
      const payment = request.headers.get('X-Payment');

      if (!payment) {
        return Response.json({
          x402Version: 1,
          error: 'Payment required',
          accepts: [{
            scheme: 'exact',
            network: process.env.STELLAR_NETWORK === 'mainnet' ? 'stellar' : 'stellar-testnet',
            maxAmountRequired: String(parseFloat(amountXlm) * 10_000_000),
            asset: 'XLM',
            payTo: process.env.PLATFORM_PUBLIC_KEY!,
          }]
        }, { status: 402 });
      }

      const valid = await verifyPayment(payment, amountXlm);
      if (!valid) return Response.json({ error: 'Invalid payment' }, { status: 402 });

      return handler(request);
    };
  };
}

// Usage in any route:
export const GET = withX402Gate('0.5')(async (request) => {
  const data = await buildAnalytics();
  return Response.json(data);
});
```

### NFC Payments

For in-person payments (cafeteria, office coffee bar):

```
1. Payee opens SocialPay, displays their QR code / NFC tag
2. Payer taps phone to NFC tag or scans QR
3. App prefills: recipient @handle, suggested amount
4. Payer confirms amount and biometric
5. Payment sent in ~5 seconds
```

### QR Code Payments

```
socialpay://send?to=@joao&amount=10&asset=USDC&description=Lunch
```

Deep link format that opens the SocialPay app (or PWA) with the send form pre-filled.

### Cross-Org Payments

Implement `@handle@company.domain` resolution via Stellar Federation:

```typescript
async function resolveExternalHandle(fullHandle: string) {
  // @joao@acme.com → joao*acme.com
  const [localPart, domain] = fullHandle.replace('@', '').split('@');
  const federationName = `${localPart}*${domain}`;

  const federationRecord = await StellarSdk.FederationServer.resolve(federationName);
  return federationRecord.account_id; // Stellar public key
}
```
