# Deployment Guide

This guide covers running Kivo Pay locally with a full stack (Go API + Supabase + Redis) and deploying to production on Fly.io (API), Cloudflare Pages (dashboard), and Supabase (database).

---

| Tool | Version | Install |

## Entry Point

```go
// cmd/server/main.go

package main

import (
    "context"
    "log"
    "os"
    "os/signal"
    "syscall"

    "github.com/gofiber/fiber/v2"
    "github.com/gofiber/fiber/v2/middleware/cors"
    "github.com/gofiber/fiber/v2/middleware/logger"
    "github.com/gofiber/fiber/v2/middleware/recover"

    "github.com/kivo-pay/kivo/internal/config"
    "github.com/kivo-pay/kivo/internal/db"
    "github.com/kivo-pay/kivo/internal/handlers"
    "github.com/kivo-pay/kivo/internal/middleware"
    "github.com/kivo-pay/kivo/internal/stellar"
    "github.com/kivo-pay/kivo/internal/workers"
)

func main() {
    ctx, cancel := signal.NotifyContext(context.Background(), syscall.SIGINT, syscall.SIGTERM)
    defer cancel()

    cfg := config.Load()

    // Database
    pool, err := config.NewPool(ctx)
    if err != nil {
        log.Fatalf("db: %v", err)
    }
    defer pool.Close()
    queries := db.New(pool)

    // Redis
    rdb := config.NewRedis(cfg.RedisURL)
    defer rdb.Close()

    // Stellar
    stellarClient := stellar.New()

    // Start background workers
    go workers.StartPaymentWorker(ctx, rdb, queries, stellarClient)
    go workers.StartWebhookWorker(ctx, rdb, queries)

    // Fiber app
    app := fiber.New(fiber.Config{
        AppName:      "Kivo Pay API v0.1.0",
        ErrorHandler: handlers.ErrorHandler,
    })

    app.Use(recover.New())
    app.Use(logger.New())
    app.Use(cors.New(cors.Config{
        AllowOrigins: cfg.CORSOrigins,
        AllowHeaders: "Origin, Content-Type, Accept, Authorization, X-API-Key, X-PAYMENT",
    }))

    // Routes
    v1 := app.Group("/v1")
    v1.Get("/health", handlers.Health(pool, rdb))

    // Device endpoints (Dashboard JWT auth)
    dash := v1.Group("/", middleware.DashboardAuth(cfg.SupabaseJWTSecret))
    dash.Post("/devices", handlers.RegisterDevice(queries, stellarClient))

    // Device-authenticated endpoints
    device := v1.Group("/", middleware.DeviceAuth(queries))
    device.Use(middleware.RateLimit(rdb, 1000, time.Minute))
    device.Get("/devices/:id",  handlers.GetDevice(queries, stellarClient))
    device.Post("/payments",    handlers.CreatePayment(queries, rdb))
    device.Get("/payments/:id", handlers.GetPayment(queries))
    device.Post("/payments/:id/execute",   handlers.ExecutePayment(queries, rdb))
    device.Post("/payments/:id/condition", handlers.SubmitCondition(queries, rdb))
    device.Get("/payments/:id/status",     handlers.GetPaymentStatus(queries, stellarClient))

    // Webhook endpoints (Dashboard JWT)
    app.Post("/v1/webhooks",     middleware.DashboardAuth(cfg.SupabaseJWTSecret), handlers.CreateWebhook(queries))
    app.Delete("/v1/webhooks/:id", middleware.DashboardAuth(cfg.SupabaseJWTSecret), handlers.DeleteWebhook(queries))

    // x402-protected endpoint example
    app.Get("/api/x402/data",
        middleware.X402(rdb, stellarClient, "0.10", "USDC:"+cfg.USDCIssuer),
        handlers.PremiumData,
    )

    // Start server
    go func() {
        port := cfg.Port
        if port == "" {
            port = "8080"
        }
        log.Printf("kivo: listening on :%s", port)
        if err := app.Listen(":" + port); err != nil {
            log.Fatalf("fiber: %v", err)
        }
    }()

    <-ctx.Done()
    log.Println("kivo: shutting down...")
    app.Shutdown()
}
```

Load in Go:

```go
// internal/config/config.go

package config

type Config struct {
    DatabaseURL             string
    RedisURL                string
    SupabaseJWTSecret       string
    StellarNetwork          string
    PlatformSecret          string
    StellarEncryptKey       string
    X402PlatformKey         string
    USDCIssuer              string
    EtherfuseMode           string
    EtherfuseBaseURL        string
    EtherfuseAPIKey         string
    EtherfuseWebhookURL     string
    EtherfuseWebhookSecret  string
    EtherfuseWebhookVerify  bool
    EtherfuseDefaultFiat    string
    EtherfuseAllowedAssets  string
    Port                    string
    CORSOrigins             string
}

func Load() *Config {
    return &Config{
        DatabaseURL:              requireEnv("DATABASE_URL"),
        RedisURL:                 requireEnv("REDIS_URL"),
        SupabaseJWTSecret:        requireEnv("SUPABASE_JWT_SECRET"),
        StellarNetwork:           getEnv("STELLAR_NETWORK", "testnet"),
        PlatformSecret:           requireEnv("PLATFORM_STELLAR_SECRET"),
        StellarEncryptKey:        requireEnv("STELLAR_ENCRYPTION_KEY"),
        X402PlatformKey:          requireEnv("X402_PLATFORM_KEY"),
        USDCIssuer:               getEnv("USDC_ISSUER", "GBBD47IF6LWK7P7MDEVSCWR7DPUWV3NY3DTQEVFL4NAT4AQH3ZLLFLA5"),
        EtherfuseMode:            getEnv("ETHERFUSE_MODE", "mock"),
        EtherfuseBaseURL:         getEnv("ETHERFUSE_BASE_URL", "https://api.sand.etherfuse.com"),
        EtherfuseAPIKey:          getEnv("ETHERFUSE_API_KEY", ""),
        EtherfuseWebhookURL:      getEnv("ETHERFUSE_WEBHOOK_URL", ""),
        EtherfuseWebhookSecret:  getEnv("ETHERFUSE_WEBHOOK_SECRET", ""),
        EtherfuseWebhookVerify:  getEnv("ETHERFUSE_WEBHOOK_VERIFY", "true") == "true",
        EtherfuseDefaultFiat:    getEnv("ETHERFUSE_DEFAULT_FIAT", "MXN"),
        EtherfuseAllowedAssets:  getEnv("ETHERFUSE_ALLOWED_ASSETS", "USDC:"+getEnv("USDC_ISSUER", "GBBD47IF6LWK7P7MDEVSCWR7DPUWV3NY3DTQEVFL4NAT4AQH3ZLLFLA5")),
        Port:                    getEnv("PORT", "8080"),
        CORSOrigins:             getEnv("CORS_ORIGINS", "http://localhost:5173"),
    }
}
```

## Docker — Multi-Stage Build

```dockerfile
# Dockerfile

# Stage 1: Build
FROM golang:1.22-alpine AS builder
WORKDIR /app
COPY go.mod go.sum ./
RUN go mod download
COPY . .
RUN CGO_ENABLED=0 GOOS=linux go build -ldflags="-s -w" -o kivo ./cmd/server/main.go

# Stage 2: Distroless runtime (minimal attack surface)
FROM gcr.io/distroless/static-debian12
COPY --from=builder /app/kivo /kivo
EXPOSE 8080
ENTRYPOINT ["/kivo"]
```

Build and run locally:

```bash
docker build -t kivo-api .
docker run -p 8080:8080 --env-file .env kivo-api
```

Docker Compose for the full local stack:

```yaml
# docker-compose.yml
version: '3.9'
services:
  api:
    build: .
    ports: ["8080:8080"]
    env_file: .env
    depends_on: [redis]

  redis:
    image: redis:7-alpine
    ports: ["6379:6379"]
```

## Fly.io Deployment

```bash
# Initial setup
fly auth login
cd apps/kivo
fly launch --name kivo-api --region iad --no-deploy
```

```toml
# fly.toml

app = "kivo-api"
primary_region = "iad"

[build]

[env]
  PORT = "8080"
  STELLAR_NETWORK = "mainnet"
  ETHERFUSE_MODE = "production"
  ETHERFUSE_BASE_URL = "https://api.etherfuse.com"
  ETHERFUSE_DEFAULT_FIAT = "MXN"

[http_service]
  internal_port = 8080
  force_https = true
  auto_stop_machines = true
  auto_start_machines = true
  min_machines_running = 0

  [http_service.concurrency]
    type = "requests"
    hard_limit = 250
    soft_limit = 200

[[vm]]
  memory = "256mb"
  cpu_kind = "shared"
  cpus = 1

[checks]
  [checks.health]
    grace_period = "10s"
    interval = "30s"
    method = "GET"
    path = "/v1/health"
    port = 8080
    timeout = "5s"
    type = "http"
```

## Free Tier Cost Breakdown

| Service | Free Tier | Kivo Usage | Cost |
|---|---|---|---|
| Fly.io | 3 shared VMs (sleep when idle) | 1 Go API instance | $0 |
| Supabase | 500 MB, 2 CPU, 50k MAU | DB + Auth | $0 |
| Upstash Redis | 10,000 commands/day | Job queue | $0 |
| Cloudflare Pages | Unlimited deployments | React dashboard | $0 |
| Stellar Testnet | Free | Development | $0 |
| Stellar Mainnet | $0.00001/tx | Per payment | ~$0 |
| **Total** | | | **$0/month** |

Scale thresholds:
- Supabase: upgrade at ~800k payment records (~$25/month for 8 GB)
- Upstash: upgrade at ~2,000 payments/day (~$0.20 per 100k commands)
- Fly.io: upgrade when sustained traffic requires non-sleeping machines
